
SELECT 
  STAT_PERIOD,  --日期
  ROUND(FIN_AWAIT,0) AS FIN_AWAIT,--理财计划待收
  FIN_AWAIT_NUM,--有待收人数
  ROUND(FIN_HUOQI_AWAIT,0) AS FIN_HUOQI_AWAIT,--活期待收总量
  ROUND(FIN_HUOQI_AWAIT/FIN_AWAIT,4) AS AWAIT_RATE, --活期待收占比
  FIN_HUOQI_AWAIT_NUM,--有活期待收人数
  ROUND(ADD_INV-SAN_INV+SAN_REPAY_CAPITAL+EXITS,0) AS FIN_INV, --理财计划投资
  ROUND(UNLOCKS,0) AS UNLOCKS,  --解锁金额  
  ROUND(APPLY_EXIT,0) AS APPLY_EXIT,  --申请退出金额
  ROUND(APPLY_EXIT_NUM,0) AS APPLY_EXIT_NUM,  --申请退出人数
  ROUND(EXITS,0) AS EXITS,  --成功退出本金
  ROUND(EXITS_ALL,0) AS EXITS_ALL,  --成功退出本息
  ROUND(EXITS/FIN_HUOQI_AWAIT,4) AS EXIT_RATE, --当日退出率
  ROUND(FIN_REPAY,0) AS FIN_REPAY, --理财计划底层回款本息
  ROUND(NO_MATCH_CAPITIL_WAIT,0) AS NO_MATCH_CAPITIL_WAIT,  --理财计划预约资金
  FIN_APR,  --理财计划加权利率
  AVG_APR  --理财计划底层资产加权利率
FROM DM_REPORT_DAILY_OPERATIONS
where 1=1
${time}
and STAT_PERIOD>20171009
